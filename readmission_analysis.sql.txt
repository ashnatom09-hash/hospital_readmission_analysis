SELECT * FROM patient LIMIT 10;
SELECT column_name, data_type
FROM information_schema.columns
WHERE table_name = 'patient';
#1What is the overall 30-day readmission rate?
SELECT 
    COUNT(*) AS total_patients,
    COUNT(*) FILTER (WHERE readmitted = 'yes') AS total_readmitted,
    ROUND(
        100.0 * COUNT(*) FILTER (WHERE readmitted = 'yes') / COUNT(*),
        2
    ) AS readmission_rate_percentage
FROM patient;

#2READMISSION BY AGE 
SELECT 
    age,
    COUNT(*) AS total_patients,
    COUNT(*) FILTER (WHERE readmitted = 'yes') AS total_readmitted,
    ROUND(
        100.0 * COUNT(*) FILTER (WHERE readmitted = 'yes') / COUNT(*),
        2
    ) AS readmission_rate_percentage
FROM patient
GROUP BY age
ORDER BY readmission_rate_percentage DESC;
